53void FToolbar::RegisterMenus()
54{
55 FToolMenuOwnerScoped OwnerScoped(this);
56
57 UToolMenu* ToolbarMenu = UToolMenus::Get()->ExtendMenu("LevelEditor.LevelEditorToolBar.PlayToolBar");
58 FToolMenuSection& Section = ToolbarMenu->FindOrAddSection("PluginTools");
59
60 const UToolbarSettings* Settings = GetDefault<UToolbarSettings>();
61 if (Settings->bEnableLevelFeature)
62 {
63 const TAttribute<FText> LabelAttr = TAttribute<FText>::CreateLambda([this]()
64 {
65 if (LevelFeature->SelectedMapPackage.IsEmpty())
66 return NSLOCTEXT("CoffeeToolbar", "Idle", "Level Selector");
67
68 const FString AssetName = FPackageName::GetLongPackageAssetName(LevelFeature->SelectedMapPackage);
69 return FText::FromString(AssetName);
70 });
71
72 FToolMenuEntry DropdownEntry = FToolMenuEntry::InitComboButton(
73 "CoffeeToolbarDropdown",
74 FUIAction(),
75 FOnGetContent::CreateRaw(LevelFeature.Get(), &FLevelFeature::GenerateLevelMenu),
76 LabelAttr,
77 LOCTEXT("CoffeeToolbarDropdown_Tooltip", "Select Level"),
78 FSlateIcon(FAppStyle::GetAppStyleSetName(), "Icons.Level")
79 );
80 DropdownEntry.StyleNameOverride = "CalloutToolbar";
81 Section.AddEntry(DropdownEntry);
82
83 Section.AddEntry(FToolMenuEntry::InitToolBarButton(
84 "CoffeeToolbar_Open",
85 FUIAction(FExecuteAction::CreateRaw(LevelFeature.Get(), &FLevelFeature::OnSelectedMap_Open)),
86 NSLOCTEXT("CoffeeToolbar","Open","Open"),
87 NSLOCTEXT("CoffeeToolbar","Open_Tip","Open The Selected Level"),
88 FSlateIcon(FAppStyle::GetAppStyleSetName(), "GraphEditor.Documentation_16x")
89 ));
90
91 Section.AddEntry(FToolMenuEntry::InitToolBarButton(
92 "CoffeeToolbar_PlaySelectedViewport",
93 FUIAction(FExecuteAction::CreateRaw(LevelFeature.Get(), &FLevelFeature::OnSelectedMap_PlaySelectedViewport)),
94 NSLOCTEXT("CoffeeToolbar","PlaySelectedViewport","Play Selected Viewport"),
95 NSLOCTEXT("CoffeeToolbar","PlaySelectedViewport_Tip","Play In The Selected Editor Viewport"),
96 FSlateIcon(FAppStyle::GetAppStyleSetName(), "Icons.Play")
97 ));
98
99 Section.AddEntry(FToolMenuEntry::InitToolBarButton(
100 "CoffeeToolbar_PlayPIE",
101 FUIAction(FExecuteAction::CreateRaw(LevelFeature.Get(), &FLevelFeature::OnSelectedMap_PlayInEditor)),
102 NSLOCTEXT("CoffeeToolbar","PlayPIE","Play In Editor"),
103 NSLOCTEXT("CoffeeToolbar","PlayPIE_Tip","Selected Editor Play In Editor"),
104 FSlateIcon(FAppStyle::GetAppStyleSetName(), "GraphEditor.PadEvent_16x")
105 ));
106 }
107
108 if (Settings->bEnableScreenshotFeature)
109 {
110 Section.AddEntry(FToolMenuEntry::InitToolBarButton(
111 "CoffeeToolbar_Screenshot",
112 FUIAction(FExecuteAction::CreateRaw(ScreenshotFeature.Get(), &FScreenshotFeature::OnCaptureScreenshot)),
113 NSLOCTEXT("CoffeeToolbar", "Screenshot", "Screenshot"),
114 NSLOCTEXT("CoffeeToolbar", "Screenshot_Tip", "Take a screenshot of the active viewport (PIE or Editor)"),
115 FSlateIconFinder::FindIconForClass(ACameraActor::StaticClass()) // ← 카메라 아이콘
116 ));
117
118 Section.AddEntry(FToolMenuEntry::InitToolBarButton(
119 "DoppleToolbar_OpenShotDir",
120 FUIAction(FExecuteAction::CreateRaw(ScreenshotFeature.Get(), &FScreenshotFeature::OnOpenScreenShotDir)),
121 NSLOCTEXT("CoffeeToolbar", "OpenShotDir", "Open Screenshot Folder"),
122 NSLOCTEXT("CoffeeToolbar", "OpenShotDir_Tip", "Open the project\'s screenshot directory"),
123 FSlateIcon(FAppStyle::GetAppStyleSetName(), "Icons.FolderOpen")
124 ));
125 }
126
127 if (Settings->bEnableCommandFeature)
128 {
129 FToolMenuEntry CommandsComboButton = FToolMenuEntry::InitComboButton(
130 "CommandsComboButton",
131 FUIAction(),
132 FOnGetContent::CreateRaw(CommandFeature.Get(), &FCommandFeature::GenerateCommandsMenu),
133 NSLOCTEXT("CoffeeToolbar", "CommandsMenu", "Commands"),
134 NSLOCTEXT("CoffeeToolbar", "CommandsMenu_Tooltip", "Execute custom commands"),
135 FSlateIcon(FAppStyle::GetAppStyleSetName(), "Icons.Settings")
136 );
137 CommandsComboButton.StyleNameOverride = "CalloutToolbar";
138 Section.AddEntry(CommandsComboButton);
139 }
140
141 UToolMenus::Get()->RefreshAllWidgets();
142}
