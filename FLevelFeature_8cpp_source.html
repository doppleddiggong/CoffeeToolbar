38bool FLevelFeature::CollectWorlds(TArray<FAssetData>& OutWorlds) const
39{
40 const TArray<FName> Roots = UToolbarSettings::GetSearchRoots();
41
42 FAssetRegistryModule& ARM = FModuleManager::LoadModuleChecked<FAssetRegistryModule>("AssetRegistry");
43 IAssetRegistry& AR = ARM.Get();
44
45#if WITH_EDITOR
46 // 에셋 레지스트리 스캔이 끝날 때까지 대기(필요 시)
47 AR.WaitForCompletion();
48#endif
49
50 FARFilter Filter;
51 Filter.bRecursivePaths = true;
52 Filter.ClassPaths.Add(UWorld::StaticClass()->GetClassPathName());
53 for (const FName& Root : Roots)
54 Filter.PackagePaths.Add(Root);
55
56 AR.GetAssets(Filter, OutWorlds);
57
58 OutWorlds.Sort([](const FAssetData& A, const FAssetData& B)
59 {
60 if (A.AssetName != B.AssetName)
61 return A.AssetName.LexicalLess(B.AssetName);
62
63 return A.PackageName.LexicalLess(B.PackageName);
64 });
65
66 return OutWorlds.Num() > 0;
67}

70bool FLevelFeature::EnsureValidSelectedMap() const
71{
72 if (SelectedMapPackage.IsEmpty())
73 {
74 UE_LOG(LogTemp, Warning, TEXT("No Level Selected!"));
75 return false;
76 }
77 if (!FPackageName::IsValidLongPackageName(SelectedMapPackage))
78 {
79 UE_LOG(LogTemp, Warning, TEXT("Invalid long package name: %s"), *SelectedMapPackage);
80 return false;
81 }
82 if (!FPackageName::DoesPackageExist(SelectedMapPackage))
83 {
84 UE_LOG(LogTemp, Warning, TEXT("Map does not exist: %s"), *SelectedMapPackage);
85 return false;
86 }
87 return true;

108TSharedRef<SWidget> FLevelFeature::GenerateLevelMenu()
109{
110 const TArray<FName> Roots = UToolbarSettings::GetSearchRoots();
111
112 FAssetRegistryModule& ARM = FModuleManager::LoadModuleChecked<FAssetRegistryModule>("AssetRegistry");
113 IAssetRegistry& AR = ARM.Get();
114
115 FARFilter Filter;
116 Filter.bRecursivePaths = true;
117 Filter.ClassPaths.Add(UWorld::StaticClass()->GetClassPathName());
118
119 for (const FName& Root : Roots)
120 Filter.PackagePaths.Add(Root);
121
122 TArray<FAssetData> Worlds;
123 AR.GetAssets(Filter, Worlds);
124
125 TMap<FName, int32> NameCount;
126 for (const FAssetData& AD : Worlds)
127 NameCount.FindOrAdd(AD.AssetName)++;
128
129 Worlds.Sort([](const FAssetData& A, const FAssetData& B)
130 {
131 if (A.AssetName != B.AssetName)
132 return A.AssetName.LexicalLess(B.AssetName);
133
134 return A.PackageName.LexicalLess(B.PackageName);
135 });
136
137 FMenuBuilder MenuBuilder(true, nullptr);
138
139 for (const FAssetData& AssetData : Worlds)
140 {
141 const FString PackagePath = AssetData.PackageName.ToString(); // "/Game/Level/DefaultMap"
142 const FString NameOnly = AssetData.AssetName.ToString(); // "DefaultMap"
143 const FString DirOnly = FPackageName::GetLongPackagePath(PackagePath); // "/Game/Level"
144 const bool bDup = NameCount[AssetData.AssetName] > 1;
145
146 const FText Label = bDup
147 ? FText::FromString(FString::Printf(TEXT("%s (%s)"), *NameOnly, *DirOnly))
148 : FText::FromString(NameOnly);
149 const FText Tooltip = FText::FromString(PackagePath);
150
151 MenuBuilder.AddMenuEntry(
152 Label, Tooltip, FSlateIcon(),
153 FUIAction(FExecuteAction::CreateRaw(this, &FLevelFeature::OnSelectedMap, PackagePath))
154 );
155 }
156
157 return MenuBuilder.MakeWidget();
158}

177void FLevelFeature::OnSelectedMap_PlaySelectedViewport()
178{
179 if (!LoadSelectedMap())
180 return;
181
182 TSharedPtr<IAssetViewport> ActiveLevelViewport;
183 if (FModuleManager::Get().IsModuleLoaded(TEXT("LevelEditor")))
184 {
185 auto& LevelEditorModule = FModuleManager::GetModuleChecked<FLevelEditorModule>(TEXT("LevelEditor"));
186 ActiveLevelViewport = LevelEditorModule.GetFirstActiveViewport();
187 }
188
189 if (GUnrealEd)
190 {
191 FRequestPlaySessionParams Params;
192 Params.WorldType = EPlaySessionWorldType::PlayInEditor;
193 Params.SessionDestination = EPlaySessionDestinationType::InProcess;
194 if (ActiveLevelViewport.IsValid())
195 Params.DestinationSlateViewport = ActiveLevelViewport;
196 GUnrealEd->RequestPlaySession(Params);
197 }
198}
