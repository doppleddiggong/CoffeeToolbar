21void FScreenshotFeature::OnCaptureScreenshot()
22{
23 // 1) PIE 중이면 게임뷰포트에 고해상도 캡쳐
24 if (GEngine && GEditor &&
25 GEditor->PlayWorld &&
26 GEngine->GameViewport &&
27 GEngine->GameViewport->Viewport)
28 {
29 FScreenshotRequest::RequestScreenshot(true);
30 GEngine->GameViewport->Viewport->TakeHighResScreenShot();
31 return;
32 }
33
34 // 2) 에디터 레벨 뷰포트가 있으면 그쪽에서 캡쳐
35 if (FLevelEditorModule* LEM = FModuleManager::GetModulePtr<FLevelEditorModule>("LevelEditor"))
36 {
37 if (TSharedPtr<IAssetViewport> AV = LEM->GetFirstActiveViewport())
38 {
39 FEditorViewportClient& VC = AV->GetAssetViewportClient();
40 VC.TakeHighResScreenShot();
41 return;
42 }
43 }
44
45 if (UWorld* W = FCommon::GetActiveTargetWorld())
46 GEditor->Exec(W, TEXT("HighResShot 1"));
47}
