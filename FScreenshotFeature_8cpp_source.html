68void FScreenshotFeature::CaptureActiveViewport(int32 ResolutionMultiplier)
69{
70 if (ResolutionMultiplier <= 0)
71 {
72 ResolutionMultiplier = 1;
73 }
74
75 if (IConsoleVariable* ResolutionCVar = IConsoleManager::Get().FindConsoleVariable(TEXT("r.HighResScreenshot.ResolutionMultiplier")))
76 {
77 ResolutionCVar->Set(ResolutionMultiplier);
78 }
79
80 if (GEngine && GEditor &&
81 GEditor->PlayWorld &&
82 GEngine->GameViewport &&
83 GEngine->GameViewport->Viewport)
84 {
85 FScreenshotRequest::RequestScreenshot(true);
86 GEngine->GameViewport->Viewport->TakeHighResScreenShot();
87 return;
88 }
89
90 if (FLevelEditorModule* LevelEditorModule = FModuleManager::GetModulePtr<FLevelEditorModule>("LevelEditor"))
91 {
92 if (TSharedPtr<IAssetViewport> AssetViewport = LevelEditorModule->GetFirstActiveViewport())
93 {
94 FEditorViewportClient& ViewportClient = AssetViewport->GetAssetViewportClient();
95 ViewportClient.TakeHighResScreenShot();
96 return;
97 }
98 }
99
100 if (UWorld* World = FCommon::GetActiveTargetWorld())
101 {
102 const FString Command = FString::Printf(TEXT("HighResShot %d"), ResolutionMultiplier);
103 GEditor->Exec(World, *Command);
104 }
105}
