98static bool NormalizeUserPath(const FString& In, FName& OutRoot)
99{
100 FString S = In;
101 S.ReplaceInline(TEXT("\\"), TEXT("/"));
102 S.TrimStartAndEndInline();
103
104 if (S.Contains(TEXT(":")))
105 return false;
106
107 if (!S.StartsWith(TEXT("/")))
108 S = TEXT("/") + S;
109
110 if (S.StartsWith(TEXT("/Content/")))
111 S = S.Replace(TEXT("/Content/"), TEXT("/Game/"));
112
113 S.RemoveFromEnd(TEXT("/"));
114 OutRoot = FName(*S);
115 return true;
116}

122TArray<FName> UToolbarSettings::GetSearchRoots(const bool bFallbackToGame)
123{
124 TArray<FName> Paths;
125 bool bAdded = false;
126
127 if (const UToolbarSettings* S = GetDefault<UToolbarSettings>()) // This line has been corrected
128 {
129 for (const FDirectoryPath& Dir : S->ExtraSearchPaths)
130 {
131 if (Dir.Path.IsEmpty())
132 continue;
133
134 FName Root;
135 if (NormalizeUserPath(Dir.Path, Root))
136 {
137 Paths.AddUnique(Root);
138 bAdded = true;
139 }
140 else
141 {
142 UE_LOG(LogTemp, Warning, TEXT("Invalid search path (cannot normalize): %s"), *Dir.Path);
143 }
144 }
145 }
146
147 if (!bAdded && bFallbackToGame)
148 Paths.Add(FName(TEXT("/Game")));
149
150 FAssetRegistryModule& ARM = FModuleManager::LoadModuleChecked<FAssetRegistryModule>("AssetRegistry");
151 IAssetRegistry& AR = ARM.Get();
152
153 for (int32 i = Paths.Num() - 1; i >= 0; --i)
154 {
155 if (!AR.PathExists(Paths[i]))
156 {
157 UE_LOG(LogTemp, Warning, TEXT("Skip invalid search path: %s"), *Paths[i].ToString());
158 Paths.RemoveAt(i);
159 }
160 }
161
162 return Paths;
163}
