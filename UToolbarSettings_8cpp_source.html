28static bool NormalizeUserPath(const FString& In, FName& OutRoot)
29{
30 FString S = In;
31 S.ReplaceInline(TEXT("\\"), TEXT("/"));
32 S.TrimStartAndEndInline();
33
34 if (S.Contains(TEXT(":")))
35 return false;
36
37 if (!S.StartsWith(TEXT("/")))
38 S = TEXT("/") + S;
39
40 if (S.StartsWith(TEXT("/Content/")))
41 S = S.Replace(TEXT("/Content/"), TEXT("/Game/"));
42
43 S.RemoveFromEnd(TEXT("/"));
44 OutRoot = FName(*S);
45 return true;
46}

52TArray<FName> UToolbarSettings::GetSearchRoots(const bool bFallbackToGame)
53{
54 TArray<FName> Paths;
55 bool bAdded = false;
56
57 if (const UToolbarSettings* S = GetDefault<UToolbarSettings>())
58 {
59 for (const FDirectoryPath& Dir : S->ExtraSearchPaths)
60 {
61 if (Dir.Path.IsEmpty())
62 continue;
63
64 FName Root;
65 if (NormalizeUserPath(Dir.Path, Root))
66 {
67 Paths.AddUnique(Root);
68 bAdded = true;
69 }
70 else
71 {
72 UE_LOG(LogTemp, Warning, TEXT("Invalid search path (cannot normalize): %s"), *Dir.Path);
73 }
74 }
75 }
76
77 if (!bAdded && bFallbackToGame)
78 Paths.Add(FName(TEXT("/Game")));
79
80 FAssetRegistryModule& ARM = FModuleManager::LoadModuleChecked<FAssetRegistryModule>("AssetRegistry");
81 IAssetRegistry& AR = ARM.Get();
82
83 for (int32 i = Paths.Num() - 1; i >= 0; --i)
84 {
85 if (!AR.PathExists(Paths[i]))
86 {
87 UE_LOG(LogTemp, Warning, TEXT("Skip invalid search path: %s"), *Paths[i].ToString());
88 Paths.RemoveAt(i);
89 }
90 }
91
92 return Paths;
93}
