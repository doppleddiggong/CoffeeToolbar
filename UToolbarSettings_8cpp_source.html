57static bool NormalizeUserPath(const FString& In, FName& OutRoot)
58{
59 FString S = In;
60 S.ReplaceInline(TEXT("\\"), TEXT("/"));
61 S.TrimStartAndEndInline();
62
63 if (S.Contains(TEXT(":")))
64 return false;
65
66 if (!S.StartsWith(TEXT("/")))
67 S = TEXT("/") + S;
68
69 if (S.StartsWith(TEXT("/Content/")))
70 S = S.Replace(TEXT("/Content/"), TEXT("/Game/"));
71
72 S.RemoveFromEnd(TEXT("/"));
73 OutRoot = FName(*S);
74 return true;
75}

81TArray<FName> UToolbarSettings::GetSearchRoots(const bool bFallbackToGame)
82{
83 TArray<FName> Paths;
84 bool bAdded = false;
85
86 if (const UToolbarSettings* S = GetDefault<UToolbarSettings>())
87 {
88 for (const FDirectoryPath& Dir : S->ExtraSearchPaths)
89 {
90 if (Dir.Path.IsEmpty())
91 continue;
92
93 FName Root;
94 if (NormalizeUserPath(Dir.Path, Root))
95 {
96 Paths.AddUnique(Root);
97 bAdded = true;
98 }
99 else
100 {
101 UE_LOG(LogTemp, Warning, TEXT("Invalid search path (cannot normalize): %s"), *Dir.Path);
102 }
103 }
104 }
105
106 if (!bAdded && bFallbackToGame)
107 Paths.Add(FName(TEXT("/Game")));
108
109 FAssetRegistryModule& ARM = FModuleManager::LoadModuleChecked<FAssetRegistryModule>("AssetRegistry");
110 IAssetRegistry& AR = ARM.Get();
111
112 for (int32 i = Paths.Num() - 1; i >= 0; --i)
113 {
114 if (!AR.PathExists(Paths[i]))
115 {
116 UE_LOG(LogTemp, Warning, TEXT("Skip invalid search path: %s"), *Paths[i].ToString());
117 Paths.RemoveAt(i);
118 }
119 }
120
121 return Paths;
122}
