15// 유틸: URL 합성 (절대 URL이면 그대로, 아니면 BaseUrl + Endpoint)
16static FString BuildFullUrl(const FString BaseUrl, const FString& Input)
17{
18 if (Input.StartsWith(TEXT("http://")) || Input.StartsWith(TEXT("https://")))
19 {
20 return Input;
21 }
22
23 if (BaseUrl.IsEmpty())
24 {
25 return Input;
26 }
27
28 // 슬래시 중복 방지
29 if (BaseUrl.EndsWith(TEXT("/")) && Input.StartsWith(TEXT("/")))
30 {
31 return BaseUrl.LeftChop(1) + Input;
32 }
33 if (!BaseUrl.EndsWith(TEXT("/")) && !Input.StartsWith(TEXT("/")))
34 {
35 return BaseUrl + TEXT("/") + Input;
36 }
37 return BaseUrl + Input;
38}

87void FNetworkTestFeature::ExecuteApiRequest(int32 ApiIndex)
88{
89 const UToolbarSettings* Settings = GetDefault<UToolbarSettings>();
90 if (!Settings || ApiIndex < 0 || ApiIndex >= Settings->NetworkTests.Num())
91 {
92 UE_LOG(LogCoffeeToolbarNetworkTest, Error, TEXT("Invalid API index: %d"), ApiIndex);
93 return;
94 }
95
96 const FString BaseUrl = Settings ? Settings->NetworkTestUrl : TEXT("http://127.0.0.1:8000");
97 const FApiSendInfo& ApiInfo = Settings->NetworkTests[ApiIndex];
98 const FString FullUrl = BuildFullUrl(BaseUrl, ApiInfo.Endpoint);
99
100 // HTTP 요청 생성
101 TSharedRef<IHttpRequest, ESPMode::ThreadSafe> Request = FHttpModule::Get().CreateRequest();
102 Request->SetURL(FullUrl);
103 Request->SetTimeout(60);
104 Request->SetHeader(TEXT("Accept"), TEXT("application/json"));
105
106 // POST 바디 저장용
107 FString JsonBodyString;
108
109 // 메서드별 설정
110 if (ApiInfo.Method == EApiHttpMethod::GET)
111 {
112 Request->SetVerb(TEXT("GET"));
113 }
114 else // POST
115 {
116 Request->SetVerb(TEXT("POST"));
117 Request->SetHeader(TEXT("Content-Type"), TEXT("application/json"));
118
119 // BodyParams를 JSON으로 변환
120 if (ApiInfo.BodyParams.Num() > 0)
121 {
122 TSharedPtr<FJsonObject> JsonObject = MakeShareable(new FJsonObject);
123 for (const auto& Pair : ApiInfo.BodyParams)
124 {
125 JsonObject->SetStringField(Pair.Key, Pair.Value);
126 }
127
128 TSharedRef<TJsonWriter<>> Writer = TJsonWriterFactory<>::Create(&JsonBodyString);
129 FJsonSerializer::Serialize(JsonObject.ToSharedRef(), Writer);
130
131 Request->SetContentAsString(JsonBodyString);
132 }
133 }
134
135 // 응답 처리
136 Request->OnProcessRequestComplete().BindLambda(
137 [ApiInfo](FHttpRequestPtr Req, FHttpResponsePtr Res, bool bSuccess)
138 {
139 if (bSuccess && Res.IsValid())
140 {
141 UE_LOG(LogCoffeeToolbarNetworkTest, Log,
142 TEXT("[Network Test] %s %s -> %d\n[RES] %s"),
143 *Req->GetVerb(),
144 *Req->GetURL(),
145 Res->GetResponseCode(),
146 *Res->GetContentAsString());
147 }
148 else
149 {
150 const int32 Code = (Res.IsValid() ? Res->GetResponseCode() : -1);
151 UE_LOG(LogCoffeeToolbarNetworkTest, Error,
152 TEXT("[Network Test] Request failed. Code=%d URL=%s"),
153 Code, Req.IsValid() ? *Req->GetURL() : TEXT("<null>"));
154 }
155 }
156 );
157
158 // 요청 로그 출력
159 if (ApiInfo.Method == EApiHttpMethod::POST && !JsonBodyString.IsEmpty())
160 {
161 UE_LOG(LogCoffeeToolbarNetworkTest, Log,
162 TEXT("[Network Test] POST %s\n[BODY] %s"),
163 *FullUrl,
164 *JsonBodyString);
165 }
166 else
167 {
168 UE_LOG(LogCoffeeToolbarNetworkTest, Log,
169 TEXT("[Network Test] %s %s"),
170 ApiInfo.Method == EApiHttpMethod::GET ? TEXT("GET") : TEXT("POST"),
171 *FullUrl);
172 }
173
174 Request->ProcessRequest();
175}
