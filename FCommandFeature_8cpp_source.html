43void FCommandFeature::OnExecuteButtonCommand(FName ButtonId)
44{
45 const UToolbarSettings* Settings = GetDefault<UToolbarSettings>();
46 const FToolbarButtonInfo* ButtonInfo = Settings->ToolbarButtons.FindByPredicate([&](const FToolbarButtonInfo& Button){ return Button.Id == ButtonId; });
47
48 if (!ButtonInfo)
49 return;
50
51 if (UWorld* W = FCommon::GetActiveTargetWorld())
52 {
53 FString Command = ButtonInfo->Command;
54 if (ButtonInfo->bIsToggleButton)
55 {
56 bool& bIsToggled = ToggleButtonState.FindOrAdd(ButtonId);
57 bIsToggled = !bIsToggled;
58
59 if (Command.Contains(TEXT(" ")))
60 {
61 Command.Append(bIsToggled ? TEXT(" 1") : TEXT(" 0"));
62 }
63 }
64 UE_LOG(LogTemp, Log, TEXT("Executing command: %s"), *Command);
65 GEditor->Exec(W, *Command);
66 }
67}
