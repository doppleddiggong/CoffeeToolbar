50static FString ResolveProjectRelative(const FString& InPath)
51{
52 FString P = InPath;
53 FPaths::NormalizeFilename(P);
54 P.TrimStartAndEndInline();
55
56 if (!FPaths::IsRelative(P))
57 return IFileManager::Get().ConvertToAbsolutePathForExternalAppForRead(*P);
58
59 const FString Full = FPaths::Combine(FPaths::ProjectDir(), P);
60 return IFileManager::Get().ConvertToAbsolutePathForExternalAppForRead(*Full);

63void FFolderFeature::OpenFolder(FString Path)
64{
65 const FString FullPath = ResolveProjectRelative(Path);
66 FString FolderToOpen = FullPath;
67
68 if (!FPaths::DirectoryExists(FolderToOpen) && FPaths::FileExists(FolderToOpen))
69 FolderToOpen = FPaths::GetPath(FolderToOpen);
70
71 if (!FPaths::DirectoryExists(FolderToOpen))
72 FPlatformProcess::ExploreFolder(*FPaths::ConvertRelativePathToFull(FPaths::ProjectDir()));
73 else
74 FPlatformProcess::ExploreFolder(*FolderToOpen);
75}
